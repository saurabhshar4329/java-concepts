import java.util.*;

class main{
  public static void main(String args[]){
   Student s1 = new Student("Saurabh", 23);
   System.out.println(s1.name);
   System.out.println(s1.enrollmenNo);
  }
}
class Student{
  String name;
  int enrollmenNo;
    
  Student(String name, int enrollmenNo){
    this.name = name;
    this.enrollmenNo = enrollmenNo;
  }
}
